create procedure ProfDetails(s_id int)
	begin
		select student_id,p.PName,p.professor_id,s.course_id
		FROM Student s
		JOIN (Professor p)
		ON s.course_id = p.course_id where s.student_id =s_id;
		END
		delimiter;

	call ProfDetails(3);
